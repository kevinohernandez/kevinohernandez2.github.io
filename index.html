  <script>

    document.addEventListener('DOMContentLoaded', function() {

      const navLabels = document.querySelectorAll('.panel-label');

      const sections = document.querySelectorAll('.content-section');

      const contentArea = document.getElementById('content');

      

      // Modal elements

      const modal = document.getElementById('imageModal');

      const modalImage = document.getElementById('modalImage');

      const modalInfo = document.getElementById('modalInfo');

      const imageCounter = document.getElementById('imageCounter');

      const closeModal = document.getElementById('closeModal');

      const prevImage = document.getElementById('prevImage');

      const nextImage = document.getElementById('nextImage');

      

      let currentGallery = [];

      let currentImageIndex = 0;

      

      // Handle navigation clicks

      navLabels.forEach(label => {

        label.addEventListener('click', function() {

          const targetId = this.getAttribute('data-target');

          const targetSection = document.getElementById(targetId);

          

          if (targetSection) {

            // Remove active class from all labels

            navLabels.forEach(l => l.classList.remove('active'));

            // Add active class to clicked label

            this.classList.add('active');

            

            // Smooth scroll to target section within the content area

            targetSection.scrollIntoView({

              behavior: 'smooth',

              block: 'start'

            });

          }

        });

      });


      // Update active navigation based on scroll position within content area

      function updateActiveNavigation() {

        const scrollPosition = contentArea.scrollTop + 50; // Offset for better detection

        

        sections.forEach((section, index) => {

          const sectionTop = section.offsetTop;

          const sectionHeight = section.offsetHeight;

          

          if (scrollPosition >= sectionTop && scrollPosition < sectionTop + sectionHeight) {

            // Remove active class from all labels

            navLabels.forEach(l => l.classList.remove('active'));

            // Add active class to corresponding label

            if (navLabels[index]) {

              navLabels[index].classList.add('active');

            }

          }

        });

      }


      // Listen for scroll events on the content area (not window) with throttling

      let ticking = false;

      function requestTick() {

        if (!ticking) {

          requestAnimationFrame(updateActiveNavigation);

          ticking = true;

          setTimeout(() => { ticking = false; }, 50); // More responsive timing

        }

      }


      // Listen to scroll events on the content area, not the window

      contentArea.addEventListener('scroll', requestTick);

      

      // Initial call to set active state

      updateActiveNavigation();


      // Gallery Modal Functionality

      function openModal(galleryType, imageIndex) {

        // Get all images from the specific gallery

        const galleryImages = document.querySelectorAll(`[data-gallery="${galleryType}"] img`);

        currentGallery = Array.from(galleryImages);

        currentImageIndex = imageIndex;

        

        if (currentGallery.length > 0) {

          updateModalImage();

          modal.classList.add('show');

          document.body.style.overflow = 'hidden'; // Prevent body scroll

        }

      }

      

      function closeModalFunction() {

        modal.classList.remove('show');

        document.body.style.overflow = ''; // Restore body scroll

      }

      

      function updateModalImage() {

        if (currentGallery[currentImageIndex]) {

          const img = currentGallery[currentImageIndex];

          modalImage.src = img.src;

          modalImage.alt = img.alt;

          imageCounter.textContent = `${currentImageIndex + 1} / ${currentGallery.length}`;

        }

      }

      

      function showNextImage() {

        currentImageIndex = (currentImageIndex + 1) % currentGallery.length;

        updateModalImage();

      }

      

      function showPrevImage() {

        currentImageIndex = (currentImageIndex - 1 + currentGallery.length) % currentGallery.length;

        updateModalImage();

      }

      

      // Event listeners for gallery items

      document.querySelectorAll('.gallery-item').forEach((item, index) => {

        item.addEventListener('click', function() {

          const galleryType = this.getAttribute('data-gallery');

          const galleryItems = document.querySelectorAll(`[data-gallery="${galleryType}"]`);

          const itemIndex = Array.from(galleryItems).indexOf(this);

          openModal(galleryType, itemIndex);

        });

      });

      

      // Modal controls

      closeModal.addEventListener('click', closeModalFunction);

      nextImage.addEventListener('click', showNextImage);

      prevImage.addEventListener('click', showPrevImage);

      

      // Click outside modal to close

      modal.addEventListener('click', function(e) {

        if (e.target === modal) {

          closeModalFunction();

        }

      });

      

      // Keyboard navigation

      document.addEventListener('keydown', function(e) {

        if (modal.classList.contains('show')) {

          switch(e.key) {

            case 'Escape':

              closeModalFunction();

              break;

            case 'ArrowRight':

              showNextImage();

              break;

            case 'ArrowLeft':

              showPrevImage();

              break;

          }

        }

      });

    });

  </script>
